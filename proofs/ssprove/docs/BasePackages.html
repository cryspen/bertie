Inductive HashFunction :=
| f_hash
| f_xtr
| f_xpd.

Definition HASH (f : HashFunction) : nat :=
  22%nat + match f with
         | f_hash ⇒ 0
         | f_xtr ⇒ 1
         | f_xpd ⇒ 2
         end.
